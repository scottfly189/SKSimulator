HKMGetCursorPos 获得鼠标坐标

功能

获得鼠标在系统中的坐标

参数

  • 参数1: 设备对象：无类型指针。使用HKMOpen可以创建它。
  • 参数2: X坐标：32位整数指针。用于接收鼠标X坐标
  • 参数3: Y坐标：32位整数指针。用于接收鼠标Y坐标

返回值

布尔值。成功返回True，失败返回False

C#例子

uint dwDevId;
IntPtr lpDev;
int x, y;
dwDevId = SkmCore.HKMSearchDevice(0x1234, 0xABCD, 0);
if (dwDevId == 0xFFFFFFFF)
{
    Console.WriteLine("未找到键鼠模拟器");
    return;
}
lpDev = SkmCore.HKMOpen(dwDevId, 0);
if (lpDev == IntPtr.Zero)
{
    Console.WriteLine("打开键鼠模拟器失败");
    return;
}
if (!SkmCore.HKMGetCursorPos(lpDev, ref x, ref y))
{
    Console.WriteLine("获得鼠标坐标失败");
    return;
}
Console.WriteLine($"鼠标坐标：{x}，{y}");
SkmCore.HKMClose(lpDev);

C语言例子

DWORD dwDevId;
LPVOID lpDev;
long x,y;
dwDevId=HKMSearchDevice(0x1234,0xABCD,0);
if(dwDevId==0xFFFFFFFF)
{
    printf("未找到键鼠模拟器\n");
    return 0;
}
lpDev=HKMOpen(dwDevId,0);
if(lpDev==NULL)
{
    printf("打开键鼠模拟器失败\n");
    return 0;
}
if(!HKMGetCursorPos(lpDev,&x,&y))
{
    printf("获得鼠标坐标失败\n");
    return 0;
}
printf("鼠标坐标：%d，%d\n",x,y);
HKMClose(lpDev);
